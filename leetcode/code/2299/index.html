    class Solution {
public:
    bool check_special_char(char c){
        string special="!@#$%^&*()-+";
        for(auto s:special){
            if(c==s){
                return true;
            }
        }
        return false;
    }
    bool strongPasswordCheckerII(string password) {
        int n=password.size();
        bool upper=false,lower=false,digit=false,special=false;
        char lastword='>';
        for(auto c:password){
            if(c==lastword)
                return false;
            else if('A'<=c&&c<='Z')
                upper=true;
            else if('a'<=c&&c<='z')
                lower=true;
            else if('0'<=c&&c<='9')
                digit=true;
            else if(check_special_char(c))
                special=true;
            lastword=c;
        }
        return n>7&&upper&&lower&&digit&&special;
    }
};
