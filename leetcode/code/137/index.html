    class Solution {
   public:
    int singleNumber(vector<int>& nums) {
        int bits[32] = {0};
        for (auto num : nums) {
            for (int i = 0; i < 32; i++) {
                bits[i] += (num >> i) & 1;
            }
        }
        int ans = 0;
        for (int i = 31; i >= 0; i--) {
            ans = (ans << 1) + (bits[i] % 3);
        }
        return ans;
    }
};
