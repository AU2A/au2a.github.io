    class Solution {
public:
    long long maximumTripletValue(vector<int>& nums) {
        long long ans=0,m;
        int n=nums.size();
        for(int i=0;i<n-2;i++){
            for(int j=i+1;j<n-1;j++){
                if(nums[i]-nums[j]>0){
                    m=0;
                    for(int k=j+1;k<n;k++){
                        m=max(m,(long long)nums[k]);
                    }
                    ans=max(ans,(long long)(nums[i]-nums[j])*m);
                }
            }
        }
        return ans;
    }
};
