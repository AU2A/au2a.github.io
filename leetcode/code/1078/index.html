    class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> words;
        string word="";
        for(auto c:text){
            if(c==' '){
                words.push_back(word);
                word="";
            }else{
                word=word+c;
            }
        }
        if(word!=""){
            words.push_back(word);
        }
        vector<string> ans;
        int n=words.size();
        for(int i=2;i<n;i++){
            if(words[i-2]==first&&words[i-1]==second){
                ans.push_back(words[i]);
            }
        }
        return ans;
    }
};
