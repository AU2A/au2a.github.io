    class Solution {
public:
    int calc(int a,int b,int c){
        if(a==c){
            return 1;
        }
        return 0;
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int n) {
        vector<int> now(8),temp(8);
        int loop=(n-1)%14;
        for(int i=0;i<8;i++){
            now[i]=cells[i];
        }
        for(int cnt=0;cnt<=loop;cnt++){
            temp[0]=0;
            for(int i=1;i<7;i++){
                temp[i]=calc(now[i-1],now[i],now[i+1]);
            }
            temp[7]=0;
            for(int i=0;i<8;i++){
                now[i]=temp[i];
            }
        }
        return now;
    }
};
