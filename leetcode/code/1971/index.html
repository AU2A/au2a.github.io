    class Solution {
   public:
    bool findPath(vector<vector<int>>& dict, vector<int>& walk, int source,
                  int destination) {
        if (source == destination) {
            return true;
        }
        walk[source] = 1;
        for (auto opt : dict[source]) {
            if (!walk[opt]) {
                if (findPath(dict, walk, opt, destination)) {
                    return true;
                }
            }
        }
        return false;
    }
    bool validPath(int n, vector<vector<int>>& edges, int source,
                   int destination) {
        if (source == destination) {
            return true;
        }
        vector<vector<int>> dict(n);
        vector<int> walk(n);
        for (auto edge : edges) {
            dict[edge[0]].push_back(edge[1]);
            dict[edge[1]].push_back(edge[0]);
        }
        // for(auto m:dict){
        //     for(auto d:m){
        //         cout<<d<<' ';
        //     }cout<<endl;
        // }
        return findPath(dict, walk, source, destination);
    }
};
