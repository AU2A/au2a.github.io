class Solution {
public:
    vector<vector<int>> dict;
    vector<int> dictCnt;
    int maxDepth(int last, int now, int ansDepth) {
        int depth = 0;
        for (auto node : dict[now]) {
            if (dictCnt[node] == 1) {
                continue;
            }
            if (node != last) {
                depth = max(depth, maxDepth(now, node, ansDepth));
                if (ansDepth < depth) {
                    return depth + 1;
                }
            }
        }
        return 1 + depth;
    }
    vector<int> findMinHeightTrees(int n, vector<vector<int>>& edges) {
        if(edges.size()==1){
            return edges[0];
        }
        dict.resize(n);
        dictCnt.resize(n);
        for (auto edge : edges) {
            dict[edge[0]].push_back(edge[1]);
            dict[edge[1]].push_back(edge[0]);
            dictCnt[edge[0]]++;
            dictCnt[edge[1]]++;
        }
        int minDepth = INT_MAX, ansDepth = INT_MAX, depth;
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            depth = 1;
            if (dictCnt[i] == 1) {
                continue;
            }
            for (auto now : dict[i]) {
                depth = max(depth, maxDepth(i, now, ansDepth));
                if (depth > ansDepth) {
                    break;
                }
            }
            if (depth == ansDepth) {
                ans.push_back(i);
            } else if (depth < ansDepth) {
                ansDepth = depth;
                ans = {i};
            }
        }
        return ans;
    }
};

class Solution {
public:
    vector<vector<int>> tree;
    vector<int> treeCnt;
    vector<int> maxDepth(vector<int> leafs) {
        vector<int> newLeafs;
        for (auto leaf : leafs) {
            for (auto node : tree[leaf]) {
                if (treeCnt[node] != 0) {
                    treeCnt[node]--;
                    if (treeCnt[node] == 1) {
                        newLeafs.push_back(node);
                    }
                }
            }
        }
        if (newLeafs.size() == 0) {
            return leafs;
        }
        return maxDepth(newLeafs);
    }
    vector<int> findMinHeightTrees(int n, vector<vector<int>>& edges) {
        if (n == 1) {
            return {0};
        }
        tree.resize(n);
        treeCnt.resize(n);
        for (auto edge : edges) {
            tree[edge[0]].push_back(edge[1]);
            tree[edge[1]].push_back(edge[0]);
            treeCnt[edge[0]]++;
            treeCnt[edge[1]]++;
        }
        vector<int> leafs;
        for (int i = 0; i < treeCnt.size(); i++) {
            if (treeCnt[i] == 1) {
                treeCnt[i]--;
                leafs.push_back(i);
            }
        }
        return maxDepth(leafs);
    }
};
