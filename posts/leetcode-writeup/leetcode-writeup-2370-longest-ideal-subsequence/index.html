class Solution {
public:
    int longestIdealString(string s, int k) {
        int n=s.size(),ans=0,temp=0;
        vector<int> seq(n);
        for(int i=0;i<n;i++){
            temp=0;
            for(int j=i-1;j>=0;j--){
                if(abs(s[i]-s[j])<=k){
                    if(seq[j]>temp){
                        temp=seq[j];
                    }
                }
            }
            seq[i]=temp+1;
            ans=max(ans,seq[i]);
        }
        return ans;
    }
};

class Solution {
public:
    int longestIdealString(string s, int k) {
        int n=s.size(),ans=0,temp=0,nowWordIdx;
        vector<int> cnt(26,0);
        for(int i=0;i<n;i++){
            temp=0;
            nowWordIdx=s[i]-'a';
            for(int j=max(nowWordIdx-k,0);j<=min(nowWordIdx+k,25);j++){
                if(cnt[j]>temp){
                    temp=cnt[j];
                }
            }
            cnt[nowWordIdx]=temp+1;
            ans=max(ans,temp+1);
        }
        return ans;
    }
};
