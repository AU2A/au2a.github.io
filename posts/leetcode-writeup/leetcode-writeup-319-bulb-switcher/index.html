bool factorIsOdd(int n){
    int cnt=0;
    for(int i=1;i<=n;i++){
        cnt+=(n%i==0);
    }
    return cnt%2==1;
}
int bulbSwitch(int n) {
    int ans=0;
    for(int i=1;i<=n;i++){
        ans+=factorIsOdd(i);
    }
    return ans;
}

int bulbSwitch(int n) {
    int ans=0,root;
    for(int i=1;i<=n;i++){
        root=sqrt(i);
        ans+=(i==root*root);
    }
    return ans;
}
